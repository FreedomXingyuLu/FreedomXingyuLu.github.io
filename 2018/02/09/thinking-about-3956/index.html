在网上查到当h总和一定时，c越大，结果越大。

通过01背包计算出每个C能得到的最大的H，复杂度∑C*N

#include <stdio.h>
#include <string.h>
#define NUM 50000
long long max(long long a,long long b)
{
 return a>b?a:b;
}
int main()
{
 long long n,N,i,j;
 long long dp[NUM],h[NUM],c[NUM];

scanf("%lld",&N);
 for(;N>0;N--)
 {
 memset(dp,0,sizeof(dp));
 scanf("%lld",&n);
 for(i=1;i<=n;i++)
 {
 scanf("%lld%lld",&h[i],&c[i]);
 for(j=NUM-1;j>=c[i];j--)
 {
 dp[j]=max(dp[j],dp[j-c[i]]+h[i]);
 }
 }
 long long ans=0;
 for(i=0;i<NUM;i++)
 ans=max(ans,dp[i]*dp[i]-dp[i]*i-i*i);
 printf("%lld\n",ans);
 }
 return 0;
}
